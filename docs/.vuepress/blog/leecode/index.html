var reverseParentheses = function(s) {
  let startPosStack = [];
  let str = s;
  let sLen = s.length;

      case "(":
        startPosStack.push(i);
        break;
      case ")":
        let positon = startPosStack.pop();
        let preStr = str.substring(0, positon);
        let nextStr = str.substring(i);
        let curStr = str
          .substring(positon, i)
          .split("")
          .reverse()
          .join("");

        break;
    }
  }
  return str.split(/[()]/g).join("");
};
