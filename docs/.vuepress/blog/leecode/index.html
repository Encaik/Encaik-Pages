var islandPerimeter = function(grid) {

  let width = grid[0].length;
  let sum = 0;

        if (indey + 1 == width || !grid[index][indey + 1]) {
          sum++;
        }

          sum++;
        }
        if (indey - 1 == -1 || !grid[index][indey - 1]) {
          sum++;
        }
      }
    }
  }
  return sum;
};
